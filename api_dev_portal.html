Integration Guide

Using Python with `requests`


import requests
import json

API_URL = "https://your-api-url.onrender.com/api/send-otp"
API_KEY = "YOUR_API_KEY" # Sent to your email
USER_EMAIL = "email_of_your_user@domain.com"

headers = {
    "Authorization": f"Bearer {API_KEY}",
    "Content-Type": "application/json"
}

payload = {
    "email": USER_EMAIL
}

try:
    response = requests.post(API_URL, headers=headers, data=json.dumps(payload))
    response.raise_for_status()  # Raise an exception for bad status codes
    
    print("Response:", response.json())
    
except requests.exceptions.RequestException as e:
    print(f"An error occurred: {e}")

Using JavaScript with `fetch`


const apiUrl = 'https://your-api-url.onrender.com/api/send-otp';
const apiKey = 'YOUR_API_KEY'; // Sent to your email
const userEmail = 'email_of_your_user@domain.com';

async function sendOtp() {
    try {
        const response = await fetch(apiUrl, {
            method: 'POST',
            headers: {
                'Authorization': `Bearer ${apiKey}`,
                'Content-Type': 'application/json'
            },
            body: JSON.stringify({ email: userEmail })
        });

        if (!response.ok) {
            throw new Error(`HTTP error! status: ${response.status}`);
        }

        const data = await response.json();
        console.log('Success:', data);

    } catch (error) {
        console.error('Error sending OTP:', error);
    }
}

sendOtp();
